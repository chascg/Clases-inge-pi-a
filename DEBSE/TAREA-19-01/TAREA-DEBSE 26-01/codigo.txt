#include <LiquidCrystal.h>
#include <Keypad.h>

LiquidCrystal lcd(4, 5, 6, 7, 8, 9);
int lectura = 0;
int frec = 1000;
int intentos = 0, teclasIngresadas = 0, intentosR = 5;

const byte filas = 4;
const byte columnas = 4;
char tecla;

// Cambiar a array de caracteres para poder modificar la contraseña
char contrasena[5] = "2707"; // 4 dígitos + terminador nulo
int x = 0;
boolean alarmaActivada = false;

// Variables para el cambio de contraseña
boolean cambioContrasena = false;
int contadorAsterisco = 0;
unsigned long tiempoUltimoAsterisco = 0;
char contrasenaActual[5] = ""; // Para verificar la contraseña actual
char nuevaContrasena[5] = "";  // Para almacenar la nueva contraseña
int etapaCambio = 0; // 0: esperando, 1: pidiendo actual, 2: verificando, 3: pidiendo nueva, 4: confirmando nueva

char Teclas[filas][columnas] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte filasPins[filas] = {A0, A1, A2, A3};
byte columPins[columnas] = {3, 2, 1, 0};
Keypad customKeypad = Keypad(makeKeymap(Teclas), filasPins, columPins, filas, columnas);

void setup() {
  pinMode(11, INPUT);   // sensor PIR
  pinMode(12, OUTPUT);  // buzzer
  pinMode(10, OUTPUT);  // led externo
  lcd.begin(16, 2);
  lcd.setCursor(0, 0);
  lcd.print("SISTEMA LISTO   ");
}

void loop() {
  lectura = digitalRead(11);
  
  tecla = customKeypad.getKey();
  
  // DETECTAR TRIPLE ASTERISCO PARA CAMBIO DE PASSWORD
  if (tecla == '*') {
    unsigned long tiempoActual = millis();
    
    // Reiniciar contador si pasó más de 3 segundos
    if (tiempoActual - tiempoUltimoAsterisco > 3000) {
      contadorAsterisco = 0;
    }
    
    contadorAsterisco++;
    tiempoUltimoAsterisco = tiempoActual;
    
    // Si se presionó '*' 3 veces
    if (contadorAsterisco == 3 && !cambioContrasena) {
      cambioContrasena = true;
      etapaCambio = 1;
      x = 0;
      contadorAsterisco = 0;
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("MODO CAMBIO PASS");
      lcd.setCursor(0, 1);
      lcd.print("Pass Actual:     ");
      delay(1000);
    }
  }
  
  // MODO CAMBIO DE CONTRASEÑA
  if (cambioContrasena) {
    if (tecla) {
      // Etapa 1: Ingresar contraseña actual
      if (etapaCambio == 1) {
        if (tecla >= '0' && tecla <= '9') {
          if (x < 4) {
            contrasenaActual[x] = tecla;
            lcd.setCursor(12 + x, 1);
            lcd.print('*');
            x++;
            
            if (x == 4) {
              contrasenaActual[4] = '\0';
              delay(500);
              
              // Verificar contraseña actual
              bool passCorrecta = true;
              for (int i = 0; i < 4; i++) {
                if (contrasenaActual[i] != contrasena[i]) {
                  passCorrecta = false;
                  break;
                }
              }
              
              if (passCorrecta) {
                // Contraseña correcta
                etapaCambio = 2;
                x = 0;
                lcd.clear();
                lcd.setCursor(0, 0);
                lcd.print("PASS CORRECTA   ");
                lcd.setCursor(0, 1);
                lcd.print("                ");
                delay(1000);
                
                // Pasar a pedir nueva contraseña
                etapaCambio = 3;
                lcd.clear();
                lcd.setCursor(0, 0);
                lcd.print("NUEVA PASSWORD: ");
                lcd.setCursor(0, 1);
                lcd.print("                ");
              } else {
                // Contraseña incorrecta
                lcd.clear();
                lcd.setCursor(0, 0);
                lcd.print("PASS INCORRECTA!");
                lcd.setCursor(0, 1);
                lcd.print("CANCELANDO...   ");
                delay(2000);
                reiniciarModoCambio();
              }
            }
          }
        }
        // Cancelar con #
        if (tecla == '#') {
          lcd.clear();
          lcd.setCursor(0, 0);
          lcd.print("CAMBIO CANCELADO");
          delay(1500);
          reiniciarModoCambio();
        }
      }
      
      // Etapa 3: Ingresar nueva contraseña
      else if (etapaCambio == 3) {
        if (tecla >= '0' && tecla <= '9') {
          if (x < 4) {
            nuevaContrasena[x] = tecla;
            lcd.setCursor(x, 1);
            lcd.print('*');
            x++;
            
            if (x == 4) {
              nuevaContrasena[4] = '\0';
              delay(500);
              
              // Pasar a confirmar nueva contraseña
              etapaCambio = 4;
              x = 0;
              lcd.clear();
              lcd.setCursor(0, 0);
              lcd.print("CONFIRMAR PASS: ");
              lcd.setCursor(0, 1);
              lcd.print("                ");
            }
          }
        }
        // Cancelar con #
        if (tecla == '#') {
          lcd.clear();
          lcd.setCursor(0, 0);
          lcd.print("CAMBIO CANCELADO");
          delay(1500);
          reiniciarModoCambio();
        }
      }
      
      // Etapa 4: Confirmar nueva contraseña
      else if (etapaCambio == 4) {
        if (tecla >= '0' && tecla <= '9') {
          if (x < 4) {
            lcd.setCursor(x, 1);
            lcd.print('*');
            
            // Verificar dígito por dígito
            if (tecla == nuevaContrasena[x]) {
              x++;
              
              if (x == 4) {
                // Contraseñas coinciden, actualizar
                for (int i = 0; i < 4; i++) {
                  contrasena[i] = nuevaContrasena[i];
                }
                contrasena[4] = '\0';
                
                lcd.clear();
                lcd.setCursor(0, 0);
                lcd.print("NUEVA PASS      ");
                lcd.setCursor(0, 1);
                lcd.print("CONFIGURADA!    ");
                delay(2000);
                reiniciarModoCambio();
              }
            } else {
              // No coinciden
              lcd.clear();
              lcd.setCursor(0, 0);
              lcd.print("PASS NO COINCIDEN");
              lcd.setCursor(0, 1);
              lcd.print("CANCELANDO...   ");
              delay(2000);
              reiniciarModoCambio();
            }
          }
        }
        // Cancelar con #
        if (tecla == '#') {
          lcd.clear();
          lcd.setCursor(0, 0);
          lcd.print("CAMBIO CANCELADO");
          delay(1500);
          reiniciarModoCambio();
        }
      }
    }
  }
  // MODO NORMAL DE ALARMA
  else {
    if (lectura == HIGH && intentos != 5) {
      lcd.setCursor(0, 0);
      lcd.print("ALARMA ACTIVADA ");
      lcd.setCursor(0, 1);
      lcd.print("-PASSWORD-      ");
      tone(12, frec);
      digitalWrite(10, HIGH);
      x = 0;
      alarmaActivada = true;
    }
    
    if (tecla && alarmaActivada) {
      teclasIngresadas++;
    }
    
    if (tecla == contrasena[x] && alarmaActivada) {
      x++;
      lcd.setCursor(0, 0);
      lcd.print("ALARMA APAGADA  ");
    }
    
    if (x == 4 || alarmaActivada == false) {
      digitalWrite(10, LOW);
      noTone(12);
      lcd.setCursor(0, 0);
      lcd.print(" PUERTA CERRADA ");
      lcd.setCursor(0, 1);
      lcd.print(" ALARMA APAGADA ");
      alarmaActivada = false;
      x = 0;
      teclasIngresadas = 0;
      intentosR = 5;
      intentos = 0;
    }
    
    if (teclasIngresadas == 4 && teclasIngresadas != 0 && x != 4 && alarmaActivada == true) {
      lcd.setCursor(0, 0);
      lcd.print("ERROR DE PASS!  ");
      lcd.setCursor(0, 1);
      lcd.print(intentosR - 1);
      lcd.print(" INTENTOS FALLIDOS");
      intentos++;
      if (x != 4)
        intentosR--;
      teclasIngresadas = 0;
    }
    
    if (intentos == 5) {
      tone(12, frec);
      lcd.setCursor(0, 0);
      lcd.print("ALARMA ACTIVADA!");
      lcd.setCursor(0, 1);
      lcd.print("SE LLAMA AL FBI!");
      digitalWrite(10, HIGH);
      delay(700);
      digitalWrite(10, LOW);
      delay(700);
      lectura = LOW;
      x = 5;
      teclasIngresadas = 5;
    }
  }
}

// Función para reiniciar el modo cambio de contraseña
void reiniciarModoCambio() {
  cambioContrasena = false;
  etapaCambio = 0;
  x = 0;
  teclasIngresadas = 0;
  contadorAsterisco = 0;
  
  // Limpiar buffers
  for (int i = 0; i < 5; i++) {
    contrasenaActual[i] = '\0';
    nuevaContrasena[i] = '\0';
  }
  
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("SISTEMA LISTO   ");
  lcd.setCursor(0, 1);
  lcd.print("                ");
}
